Node.js Example

const crypto = require('crypto');

function sign(secret, timestamp, method, path, body = '') {
  const message = `${timestamp}|${method}|${path}|${body}`;
  return crypto.createHmac('sha256', secret).update(message).digest('hex');
}

async function createOrder(apiKey, apiSecret, payload) {
  const path = '/api/v1/orders/create';
  const body = JSON.stringify(payload);
  const ts = Math.floor(Date.now() / 1000).toString();
  const sig = sign(apiSecret, ts, 'POST', path, body);

  const res = await fetch('https://upays.in' + path, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-Merchant-Key': apiKey,
      'X-Timestamp': ts,
      'X-Signature': sig,
    },
    body,
  });
  return res.json();
}

// Usage
const data = await createOrder(process.env.UPIPAYS_KEY, process.env.UPIPAYS_SECRET, {
